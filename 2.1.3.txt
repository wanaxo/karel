void cleanTheTunnels(){
    while (onBeeper() || beeperAhead()){
        clearATunnel();
    }
}

void clearATunnel(){
    turnLeft();
    while (onBeeper() && beeperAhead()){
        pickBeeper();
        moveForward();
    }
    if (onBeeper() && !beeperAhead()){
        pickBeeper();
        turnAround();
    }
    while (frontIsClear()){
        moveForward();
    }
    turnLeft();
    if (frontIsClear()){
        moveForward();
    }
}
